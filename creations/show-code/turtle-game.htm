import turtle

turtle.bgcolor('black')
turtle.pencolor('lightgreen')
turtle.color('lightgreen')

turtle.lt(90)
def ford():
    turtle.fd(100)
def back():
    turtle.bk(100)
def right():
    turtle.rt(90)
def left():
    turtle.lt(90)

turtle.onkeypress(ford,"Up")
turtle.onkeypress(back,"Down")
turtle.onkeypress(right,"Right")
turtle.onkeypress(left,"Left")

turtle.listen()
turtle.mainloop()
